#Criando a REDE QUE EU ESQECI dessa porra Afff e colocar o postgres nela isso pq era a primeira coisa no docker-compose

docker network create tipsbank
docker network connect tipsbank tipsbank-postgres 

#API-CONTAS - referencia docker-compose.yaml

docker run -d \
--name tipsbank-api-contas \
--network tipsbank \
-e DB_URL="postgresql+psycopg://tipsbank:tipsbank@tipsbank-postgres:5432/tipsbank" \
-e LOG_LEVEL=INFO \
-p 8081:8080
tipsbank-api-contas:1.0
